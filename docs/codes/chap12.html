def lr_test(restr, full):
    from scipy import stats
    lr_df = (restr.df_resid - full.df_resid)
    lr_stat = -2*(restr.llf - full.llf)
    lr_pvalue = stats.chi2.sf(lr_stat, df=lr_df)
    return {"lr": lr_stat, "pvalue": lr_pvalue, "df": lr_df}
